%% Solving Example 1 from the texbook p651

% initial guess
x = [0.1,0.1,-0.1]';

% Tolerance and maximum iterations
TOL = 1e-6;
N = 7;

% Broyden's method
A = J(x);
v = F(x);

A = inv(A);

s = - A*v;
x = x + s;
k = 2;

success = 0;

while k <= N

    w = v;
    v = F(x);
    y = v - w;
    
    z = -A*y;
    
    p = - s' * z;
    
    u = s' * A;
    
    tmp = (s + z) * u;
    A = A + 1/p * tmp;
    
    s = - A*v;
    
    x = x + s;
    
    if norm(s) < TOL
        success = 1;
        break
    end
    k = k + 1;
end

if success == 1
    fprintf("The procedure was successful.\n")
else
    fprintf("The procedure was unsuccessful.\n")
end
